int test(int i, int j, int k) {
	while (i < j) {
		int mid;
		mid = (i + j) / 2;
		if (mid == k) {
			return mid;
		} 
		if (mid < k) {
			i = mid + 1;
		} else {
			j = mid - 1;
		}
	}
	return i;
}

int main() {
	int a;
	int b;
	int c;
	a = 1;
	b = 100;
	c = 66;
	int d;
	d = test(a, b, c);
	return 0;
}
